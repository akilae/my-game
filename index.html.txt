```html
<!DOCTYPE html>
<html lang="zh-TW">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0, maximum-scale=1.0, user-scalable=no">
    <title>霓虹消磚塊 - 我的作品</title>
    <script src="https://cdn.tailwindcss.com"></script>
    <script type="module">
        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";
        import { getFirestore, collection, addDoc, onSnapshot, query } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";
        import { getAuth, signInAnonymously, onAuthStateChanged } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js";

        // ==========================================
        // ⚠️ 請在此處貼上你從 Firebase 複製的設定碼 ⚠️
        // ==========================================
        const firebaseConfig = {
            apiKey: "你的_API_KEY",
            authDomain: "你的_PROJECT_ID.firebaseapp.com",
            projectId: "你的_PROJECT_ID",
            storageBucket: "你的_PROJECT_ID.appspot.com",
            messagingSenderId: "你的_SENDER_ID",
            appId: "你的_APP_ID"
        };
        // ==========================================

        const app = initializeApp(firebaseConfig);
        const db = getFirestore(app);
        const auth = getAuth(app);

        // 初始化 Firebase 功能
        onAuthStateChanged(auth, (user) => {
            if (user) {
                const leaderboardRef = collection(db, 'leaderboard');
                onSnapshot(query(leaderboardRef), (snapshot) => {
                    const scores = [];
                    snapshot.forEach((doc) => scores.push(doc.data()));
                    scores.sort((a, b) => b.score - a.score);
                    const list = document.getElementById("leaderboardList");
                    list.innerHTML = scores.slice(0, 5).map((s, i) => `
                        <div class="flex justify-between bg-slate-800/50 p-3 rounded-lg border border-white/5">
                            <span class="text-cyan-400 font-bold">#${i+1} ${s.name}</span>
                            <span class="text-white font-mono">${s.score}</span>
                        </div>
                    `).join('') || "<div class='text-slate-500'>暫無紀錄</div>";
                    document.getElementById("loadingLeaderboard").classList.add("hidden");
                });
            } else {
                signInAnonymously(auth);
            }
        });

        // 遊戲存分功能
        window.saveScore = async () => {
            const name = document.getElementById("playerName").value.trim() || "無名英雄";
            const score = parseInt(document.getElementById("finalScore").innerText);
            await addDoc(collection(db, 'leaderboard'), { name, score, date: new Date().toISOString() });
            location.reload();
        };
    </script>
    <style>
        body { touch-action: none; overflow: hidden; background-color: #020617; color: white; }
        canvas { display: block; touch-action: none; }
        .neon-shadow { text-shadow: 0 0 10px rgba(34, 211, 238, 0.8); }
    </style>
</head>
<body class="flex items-center justify-center min-h-screen">

    <!-- 頂部資訊列 -->
    <div class="absolute top-6 left-0 right-0 flex justify-around z-10 pointer-events-none">
        <div class="text-center"><p class="text-xs text-cyan-500 uppercase font-bold">Score</p><p id="scoreText" class="text-2xl font-black text-white">0</p></div>
        <div class="text-center"><p class="text-xs text-yellow-500 uppercase font-bold">Health</p><p id="livesText" class="text-2xl text-yellow-400">❤❤❤</p></div>
    </div>

    <canvas id="gameCanvas"></canvas>

    <!-- 開始/結束選單 -->
    <div id="overlay" class="fixed inset-0 bg-slate-950/90 flex flex-col items-center justify-center z-20 px-6 backdrop-blur-md">
        <div id="startUI" class="text-center w-full max-w-sm">
            <h1 class="text-6xl font-black text-cyan-400 mb-8 italic neon-shadow">NEON STRIKE</h1>
            <button onclick="startGame()" class="w-full py-4 rounded-2xl bg-cyan-500 text-slate-950 font-black text-xl hover:scale-105 transition-transform">開始遊戲</button>
            <div class="mt-8 bg-slate-900/50 p-4 rounded-2xl border border-white/5">
                <p class="text-cyan-500 text-xs font-bold uppercase tracking-widest mb-4">世界排行榜</p>
                <div id="loadingLeaderboard" class="text-slate-500 text-sm italic">載入中...</div>
                <div id="leaderboardList" class="space-y-2 text-sm"></div>
            </div>
        </div>

        <div id="gameOverUI" class="text-center hidden w-full max-w-sm">
            <h2 class="text-pink-500 font-bold tracking-widest mb-2">GAME OVER</h2>
            <p id="finalScore" class="text-7xl font-black mb-10">0</p>
            <input type="text" id="playerName" maxlength="8" placeholder="輸入你的暱稱" class="w-full bg-slate-900 border border-slate-700 p-4 rounded-xl mb-4 text-center text-white outline-none focus:border-cyan-500">
            <button onclick="saveScore()" class="w-full py-4 rounded-xl bg-cyan-500 text-slate-950 font-bold mb-3">上傳並返回</button>
            <button onclick="location.reload()" class="w-full py-4 rounded-xl bg-slate-800 text-white font-bold">直接重來</button>
        </div>
    </div>

    <script>
        const canvas = document.getElementById("gameCanvas");
        const ctx = canvas.getContext("2d");
        const overlay = document.getElementById("overlay");
        let score = 0, lives = 3, running = false;
        let paddleW = 100, paddleX, ballX, ballY, dx = 4, dy = -4;
        let bricks = [];

        function initBricks() {
            bricks = [];
            const cols = Math.floor(canvas.width / 75);
            for(let c=0; c<cols; c++) {
                bricks[c] = [];
                for(let r=0; r<4; r++) bricks[c][r] = { x:0, y:0, status: 1 };
            }
        }

        function startGame() {
            canvas.width = window.innerWidth;
            canvas.height = window.innerHeight;
            paddleX = (canvas.width - paddleW) / 2;
            ballX = canvas.width / 2;
            ballY = canvas.height - 100;
            initBricks();
            score = 0; lives = 3;
            running = true;
            overlay.style.display = 'none';
            draw();
        }

        function draw() {
            if(!running) return;
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            
            // 畫磚塊
            let activeBricks = 0;
            bricks.forEach((col, i) => col.forEach((b, j) => {
                if(b.status === 1) {
                    activeBricks++;
                    b.x = i * 75 + 10; b.y = j * 30 + 100;
                    ctx.fillStyle = `hsl(${i * 40}, 70%, 50%)`;
                    ctx.fillRect(b.x, b.y, 65, 20);
                    if(ballX > b.x && ballX < b.x+65 && ballY > b.y && ballY < b.y+20) {
                        dy = -dy; b.status = 0; score += 10;
                        document.getElementById("scoreText").innerText = score;
                    }
                }
            }));

            if(activeBricks === 0) initBricks();

            // 畫球與擋板
            ctx.fillStyle = "white"; ctx.beginPath(); ctx.arc(ballX, ballY, 8, 0, Math.PI*2); ctx.fill();
            ctx.fillStyle = "#22d3ee"; ctx.fillRect(paddleX, canvas.height-40, paddleW, 10);

            // 碰撞檢測
            if(ballX + dx > canvas.width || ballX + dx < 0) dx = -dx;
            if(ballY + dy < 0) dy = -dy;
            else if(ballY + dy > canvas.height - 40) {
                if(ballX > paddleX && ballX < paddleX + paddleW) dy = -dy;
                else if(ballY > canvas.height) {
                    lives--;
                    document.getElementById("livesText").innerText = "❤".repeat(lives);
                    if(lives <= 0) {
                        running = false;
                        document.getElementById("finalScore").innerText = score;
                        document.getElementById("startUI").classList.add("hidden");
                        document.getElementById("gameOverUI").classList.remove("hidden");
                        overlay.style.display = 'flex';
                    } else {
                        ballX = canvas.width/2; ballY = canvas.height-100; dy = -4;
                    }
                }
            }
            ballX += dx; ballY += dy;
            requestAnimationFrame(draw);
        }

        // 觸控控制
        canvas.addEventListener("touchmove", (e) => {
            paddleX = e.touches[0].clientX - paddleW / 2;
            e.preventDefault();
        }, {passive: false});
    </script>
</body>
</html>
```
